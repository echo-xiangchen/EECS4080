-- test iff with initialized boolean variable declaration with relational value (no error)
int i = 1 + 3
int j = i * 2
int m = 3 * 4 + 5
boolean p = i > 0
boolean q = j >= i
verify p <=> (q <=> not p)
boolean r = j >= m
boolean s = j < 20
verify p <=> (q <=> r) <=> not s
verify q <=> (not p <=> r and s)
