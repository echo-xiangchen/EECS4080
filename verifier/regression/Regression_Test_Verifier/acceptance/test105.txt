-- test propositional logic 
-- with initialized boolean variable declaration with relational value + propositional value (no error)
int i = 1 + 3
int j = i * 2
int m = 3 * 4 + 5
boolean p = not i > 0
boolean q = i > j=> m = i * j
verify not p => q and p
boolean r = j >= m and p
boolean s = j < 20 or m = 17
verify p <=> (not q and r or s) => not s
boolean s1 = j = i => s
boolean r0 = i = j <=> s1
boolean k = m < i and not r
verify not r0 and s1 => not (k or r)
