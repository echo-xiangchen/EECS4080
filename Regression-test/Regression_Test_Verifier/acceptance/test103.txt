-- test propositional logic 
-- with initialized boolean variable declaration with relational value (no error)
int i = 1 + 3
int j = i * 2
int m = 3 * 4 + 5
boolean p = i > 0
boolean q = j >= i
verify not p or q => q
boolean r = j >= m
boolean s = j < 20
verify r => s and (not p <=> q)
verify not s <=> (p and (q => r or q))
