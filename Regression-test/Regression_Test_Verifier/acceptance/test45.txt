-- test disjunction 
-- with initialized boolean variable declaration with relational value + propositional value (no error)
int i = 1 + 3
int j = i * 2
int m = 3 * 4 + 5
boolean p = not i > 0
boolean q = i > j or m = i * j
verify p or q
boolean r = j >= m and not p
boolean s = j < 20 or m = 17
verify not r or s or q
boolean s1 = j = i => s
boolean r0 = i = j <=> s1
boolean k = m < i and not r
verify not r0 or s1 or not k or r
