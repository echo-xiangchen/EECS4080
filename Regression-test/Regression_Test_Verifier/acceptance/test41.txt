-- test disjunction 
-- with initialized boolean variable declaration with propositional value (no error)
boolean p = true
boolean q = not p and p
verify q or not p
boolean r = p and q or not p
boolean t1 = p or q
verify r or t1 or not p or q
verify t1 or r
boolean t2 = t1 => q
boolean t3 = p <=> t2
verify t3 or t2 or not t1
