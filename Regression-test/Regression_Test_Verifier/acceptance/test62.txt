-- test iff 
-- with initialized boolean variable declaration with propositional value (no error)
boolean p = true
verify q <=> not p
boolean q = not p <=> p
boolean r = p => q
boolean t1 = p or q
verify r <=> (t1 <=> not p) <=> q
verify t1 <=> (r <=> not p)
boolean q = t1 <=> p and r
boolean t2 = t1 => q
boolean t3 = p <=> t2
verify (t3 <=> t2) <=> not t1
