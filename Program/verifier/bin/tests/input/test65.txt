-- test iff  with initialized boolean variable declaration with relational value + propositional value (no error)
int i = 1 + 3
int j = i - 20
int m = 3 * 16 + 5
boolean p = not i > 0
boolean q = i > j or m = i * j
verify p <=> q
boolean r = j >= m and not p
boolean s = j < 20 or m = 17
verify not r <=> (s <=> q) <=> p
boolean s1 = j = i or s
boolean r0 = i = j <=> s1
boolean k = m < i => not r
verify not r0 <=> s1 <=> (not k <=> r)
